SELECT 
       F.VALEREFEICFUNC,F.DTTRANSFFUNC, F.CODIGOEMPRESA EMPRESA,
       F.CODINTFUNC,
       F.CODFUNC AS CRACHA,
       F.CHAPAFUNC AS CHAPA,
       F.NOMEFUNC AS NOME,
       D1.MAE,
       (SELECT D.NRDOCTO FROM FLP_DOCUMENTOS D WHERE F.CODINTFUNC = D.CODINTFUNC AND D.TIPODOCTO = 'CPF') AS CPF, 
       F.DESCFUNCAO AS FUNCAO,
       F.DESCDEPTO AS DEPARTAMENTO,
       F.DESCAREA AS AREA,
       F.DESCSECAO,
       F.DESCSETOR,
       F.ENDRUAFUNC AS ENDERECO,
       F.ENDNRFUNC AS CASA,
       F.ENDBAIRROFUNC AS BAIRRO,
       F.ENDCIDADEFUNC AS CIDADE,
       F.FONEFUNC,
       F.FONE2FUNC,
       F.DTADMFUNC AS ADMISSAO, 
       F.SITUACAOFUNC AS SITUACAO,
       F.SALBASE,
       F.SALAUX1,
       F.SALAUX2,
       Q.DTCOMPETQUITA,
       Q.IDQUITA,
       Q.DTDESLIGQUITA  
         
FROM VW_FUNCIONARIOS F,
     VW_QUITACAOATIVA Q,
     (SELECT D.CODINTFUNC, D.NOMEDEPEN AS MAE
FROM FLP_DEPENDENTES D 
WHERE D.CODPAREN = 10)D1

WHERE F.CODINTFUNC = Q.CODINTFUNC (+)
  AND F.CODINTFUNC = D1.CODINTFUNC (+)
  AND F.SITUACAOFUNC IN ('A', 'F', 'D')
  AND F.CODIGOEMPRESA IN(4)
-- and f.coddepto in(3000)
-- AND F.CODFUNC BETWEEN 23029 AND 23044
-- AND F.CODFUNC BETWEEN 0 AND 300000
-- AND CODFUNC = 10170
-- AND F.DTADMFUNC BETWEEN '01-MAY-2014' AND '05-AUG-2014'
-- and f.VALEREFEICFUNC = 'N'

ORDER BY  F.CODIGOEMPRESA, F.CODFUNC
